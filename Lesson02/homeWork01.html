c = ++a; - эквивалентно a = a + 1; c = a; - сначала увеличиваем значение "a" на 1 (++ перед переменной) - получаем 2, значит "с" получает значение 2

d = b++; - эквивалентно d = b; b = b + 1; - сначала присвоим "d" значение "b" (++ после переменной), потом увеличиваем значение "b" на 1 - получаем 2

c = (2+ ++a); - эквивалентно a = a + 1; c = 2 + a; - "a" равно 2, увеличиваем его на единицу (++a) - получаем 3, и прибавляем 3 к 2: "c" равно 5

d = (2+ b++); - эквивалентно d = 2 + b; b = b + 1; - "b" равно 2, добавляем к этому 2 - получаем "d" равное 4, после чего прибавляем 1 к "b" (b++)
